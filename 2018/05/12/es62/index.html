5.对象代理

什么是对象代理？比如我们封装了一组数据，通过访问代理，对数据访问做限制，而用户访问的是代理层，而不是源数据。这样就进行数据保护。

ES3/ES5 和ES6数据保护区别

//ES5写法
//ES5
var Person = {
name: ‘阿蔡’,
age: 23
}
Object.defineProperty(Person,’sex’,{
writable:false,
value:’female’
})
console.table({
name: Person.name,
sex:Person.sex,
age: Person.age
})

let Person = {
name: ‘阿蔡’,
sex: ‘女’,
age: 23
}
let person = new Proxy(Person,{ //Person为代理的对象
//get为读取操作,参数target为代理对象的数据，key是你要读的哪个属性。
get:function(target, key){
return target[key]
},
//set为设置修改操作，value为属性值
set:function(target,key,value){
if(key!==’sex’){
target[key]=value;
}
}
})
console.table({
name:person.name,
sex:person.sex,
age:person.age
});
`
